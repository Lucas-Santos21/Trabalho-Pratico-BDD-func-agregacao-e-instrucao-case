R: update notas set vl_media =
	CASE
		when vl_media - floor(vl_media) <= 0.25 then floor(vl_media)
		when vl_media - floor(vl_media) >= 0.75 then floor(vl_media) + 1
		else floor(vl_media) + 0.5
	end
